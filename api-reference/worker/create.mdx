---
title: "Create Worker"
api: "POST https://sandbox.api.scratchie.com/worker/create"
description: "The Create Worker API allows you to add a new worker to the Scratchie Open API system and associate them with a project."
---

### Body Parameters

<ParamField body="FirstName" type="String">
  The first name of the user.
</ParamField>

<ParamField body="LastName" type="String">
  The last name of the user.
</ParamField>

<ParamField body="MobileNo" type="Number">
  The mobile number of the worker.
</ParamField>

<ParamField body="Email" type="String">
  The unique email of the worker.
</ParamField>

<ParamField body="Password" type="String">
  The password of the worker.
</ParamField>

<ParamField body="Worker" type="Boolean">
  If you are creating a worker, set the value of Worker to true. Otherwise, when creating an award-giver action, set Worker to false.
</ParamField>

<ParamField body="ProjectId" type="Number">
  The ID of the project associated with the worker.
</ParamField>

<ParamField body="subcontractorId" type="Number">
  The ID of the subcontractor associated with the worker.
</ParamField>

### Headers

<ParamField body="Authorization" type="String">
  Bearer `<access_token>`
</ParamField>

### Response

<ResponseField name="status" type="Number">
  The HTTP status code of the response.
</ResponseField>

<ResponseField name="err" type="Number">
  An error code or indicator (0 if successful).
</ResponseField>

<ResponseField name="data" type="Object">
  Information about the created worker.

  <ResponseField name="RoleId" type="Number">
    The role ID of the worker (5 for worker).
  </ResponseField>

  <ResponseField name="worker" type="Number">
    Indicator (1 for worker).
  </ResponseField>

  <ResponseField name="ReferralCode" type="String">

  <ResponseField name="StripAccountId" type="String">

  <ResponseField name="BankAccountToken" type="String">
