### Request Parameters

<ParamField body="workerId" type="Number">
  This field represents the ID of the worker you want to update, obtained from the previous worker creation endpoint.
</ParamField>

<ParamField body="account_type" type="String">
  Only expect 'savings' or 'checking'.
</ParamField>

<ParamField body="holder_type" type="String">
  Only expect 'business' or 'personal'.
</ParamField>

<ParamField body="bank_name" type="String">
  User Bank name.
</ParamField>

<ParamField body="account_name" type="String">
  User Bank account holder name.
</ParamField>

<ParamField body="account_number" type="Number">
  User Bank account number.
</ParamField>

<ParamField body="routing_number" type="Number">
  Routing number / BSB number.
</ParamField>

### Headers

<ParamField body="Content-Type" type="String">
  application/json
</ParamField>

<ParamField body="Authorization" type="String">
  Bearer `<access_token>`
</ParamField>
