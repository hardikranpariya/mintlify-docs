---
title: "Show All Projects Of Builder"
api: "POST https://sandbox.api.scratchie.com/project/listall"
description: "The List All Projects Of The Builder API allows you to retrieve all the projects associated with a specific builder within the Scratchie Open API system."
---

### Body Parameters

<ParamField body="userId" type="Number">
  The ID of the builder whose projects you want to retrieve.
</ParamField>

<ParamField body="start" type="Number">
  Where to start the record (Default = 1).
</ParamField>

<ParamField body="limit" type="Number">
  Number of records to retrieve (Default = 10).
</ParamField>

### Headers

<ParamField body="Authorization" type="String">
  Bearer `<access_token>`
</ParamField>

### Response

<ResponseField name="status" type="Number">
  The HTTP status code of the response.
</ResponseField>

<ResponseField name="err" type="Number">
  An error code or indicator (0 if successful).
</ResponseField>

<ResponseField name="data" type="Array">
  An array of project objects associated with the builder.

  <ItemField type="Object">
    <ItemField name="Id" type="Number">
      The project's unique ID.
    </ItemField>

    <ItemField name="Name" type="String">
      The name of the project.
    </ItemField>

    <ItemField name="BuilderProjectReferenceNumber" type="String">
      The unique reference number assigned to the project by the builder.
    </ItemField>

    <ItemField name="ScratchieProjectNumber" type="String">
      The project's unique Scratchie project number.
    </ItemField>

    <ItemField name="Address" type="String">
      The project's physical address.
    </ItemField>

    <ItemField name="MobileNo" type="String">
      The mobile number associated with the project (or null if not provided).
    </ItemField>

    <ItemField name="AssignProjectCaId" type="Number">
      ID of the assigned project (if any).
    </ItemField>

    <ItemField name="FundingOptions" type="String">
      Funding options for the project.
    </ItemField>

    <ItemField name="minimumAmount" type="Number">
      The required wallet size for the project, which must exceed $2,000.
    </ItemField>

    <ItemField name="lastMonthCarriedOver" type="Number">
      The carried-over amount from the last month (if any).
    </ItemField>

    <ItemField name="PointPlan" type="Number">
      The point plan for the project.
    </ItemField>

    <ItemField name="CreatedBy" type="Number">
      The ID of the user who created the project.
    </ItemField>

    <ItemField name="CreatedOn" type="String">
      The date and time when the project was created.
    </ItemField>

    <ItemField name="UpdatedBy" type="Number">
      The ID of the user who last updated the project.
    </ItemField>

    <ItemField name="UpdatedOn" type="String">
      The date and time when the project was last updated.
    </ItemField>

    <ItemField name="Deleted" type="Number">
      Deletion status (0 if not deleted).
    </ItemField>

    <ItemField name="DeleteBy" type="String">
      The user who deleted the project (or null if not deleted).
    </ItemField>

    <ItemField name="zaiWalletPayId" type="String">
      Zai wallet payment ID.
    </ItemField>

    <ItemField name="zaiUserId" type="String">
      Zai user ID.
    </ItemField>

    <ItemField name="awardGiverWin" type="Number">
      Award giver status.
    </ItemField>

    <ItemField name="procoreCompanyId" type="String">
      Procore company ID (or null if not provided).
    </ItemField>

    <ItemField name="scratchieOpenApi" type="Number">
      Scratchie Open API status (1 if enabled).
    </ItemField>
  </ItemField>

</ResponseField>

<ResponseField name="msg" type="String">
  A message indicating the result of the operation.
</ResponseField>

```bash
curl --location --request POST 'https://{BASE_URL}/project/listall' \
--header 'Content-Type: application/json' \
--header 'Authorization: Bearer <access_token>' \
--data-raw '{
    "userId": 2
}'
```
</RequestExample>
