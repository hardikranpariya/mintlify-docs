### Request Parameters

<ParamField body="caId" type="Number">

</ParamField>

<ParamField body="firstDefaultId" type="Number">
  This field represents the Id of the first default award category.
</ParamField>

<ParamField body="secoundDefaultId" type="Number">
  This field represents the Id of the second default award category.
</ParamField>

<ParamField body="thirdDefaultId" type="Number">
  This field represents the Id of the third default award category.
</ParamField>

<ParamField body="fourthDefaultId" type="Number">
  This field represents the Id of the fourth default award category.
</ParamField>

<ParamField body="fifthDefaultId" type="Number">
  This field represents the Id of the fifth default award category.
</ParamField>
