---
title: "Show CA Default Category"
api: "GET https://sandbox.api.scratchie.com/award-default_category/:caId"
description: "The Get CA Default category API allows you to retrieve detailed information about a specific CA Default category within Scratchie open API system."
---

### Request Parameters

<ParamField path="caId" type="Number">
  This field represents the Id of the Contracts Administrator (CA) for which you want to retrieve the default award categories.
</ParamField>

### Headers

<ParamField body="Authorization" type="String">
  Bearer `<access_token>`
</ParamField>

### Response

<ResponseField name="status" type="Number">
  The HTTP status code of the response.
</ResponseField>

<ResponseField name="err" type="Number">
  An error code or indicator (0 if successful).
</ResponseField>

<ResponseField name="data" type="Array">
  An array of default award category objects.

  <SubField name="Id" type="Number">
    The unique identifier of the award category.
  </SubField>

  <SubField name="Name" type="String">
    The name of the award category.
  </SubField>

  <SubField name="Image" type="String">
    URL of the award category image.
  </SubField>

  <SubField name="AwardTag" type="String">
    Tags associated with the award category.
  </SubField>

  <SubField name="CreatedOn" type="String">
    Date and time when the award category was created.
  </SubField>

  <SubField name="CreatedBy" type="Number">
    The user who created the award category.
  </SubField>

  <SubField name="UpdatedOn" type="String">
    Date and time when the award category was last updated.
  </SubField>

  <SubField name="UpdatedBy" type="Number">
    The user who last updated the award category.
  </SubField>

  <SubField name="Deleted" type="Number">
    Indicates if the award category is deleted (0 if not deleted).
  </SubField>

  <SubField name="DeleteBy" type="Number">
    The user who deleted the award category (0 if not deleted).
  </SubField>
</ResponseField>

<ResponseField name="msg" type="String">
  A message indicating the result of the operation.
</ResponseField>

<RequestExample>

```http
GET https://{BASE_URL}/award-default_category/12
Authorization: Bearer <access_token>
```
